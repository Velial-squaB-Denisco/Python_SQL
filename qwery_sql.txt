Создание таблицы -

CREATE TABLE books (
    book_id INT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    author VARCHAR(255) NOT NULL,
    publication_year INT,
    genre VARCHAR(50)
);

Добавление колонок -

ALTER TABLE books
ADD COLUMN isbn VARCHAR(13);

Заполнение колонок -

Удаление таблицы -

DROP TABLE имя_таблицы;


Изменение типа данных таблицы
